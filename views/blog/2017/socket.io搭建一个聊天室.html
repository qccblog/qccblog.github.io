    # 搭建一个简单的实时聊天项目

    使用express快速搭建web服务器，使用socket.io处理实时聊天。实现上参考了Node.js + Web Socket 打造即时聊天程序嗨聊 (opens new window)。
    初始化一个node项目，再安装express和socket.io，为了方便还安装了path和ejs。

    var express = require('express');
var path = require('path');
var app = express();

var http = require('http').Server(app);
var io = require('socket.io')(http);

app.set('views', path.join(__dirname, 'views'));
app.set('view engine', 'ejs');

app.use(express.static(path.join(__dirname, 'public')));

app.get('/', function(req, res){
  res.render('index');
});

io.on('connection', function(socket){
  var id = socket.id;
  var sockets = io.sockets.sockets;
  socket.on('login', function (msg) { //监听登录事件
    if (msg.length < 1) {
      return socket.emit('loginFail', '请填写昵称！');  //告知登录结果
    }
    if (socket_users.indexOf(msg) > -1) {
      socket.emit('loginFail', '昵称已存在！');
    } else {
      socket.nickname = msg;
      socket.emit('loginSuccess');
      io.sockets.emit('systemMsg', msg + '加入房间！'); //广播通知全体人员新用户加入
    }
  });
  socket.on('postMsg', function (msg) { //监听发送消息事件
    if (!socket.nickname) {
      socket.emit('loginFail', '请先登录！');
    } else {
      io.sockets.emit('newMsg', socket.nickname, msg);
    }
  });
  socket.on('disconnect', function() {  //监听断线事件
    if (socket.nickname) {  //重复刷新页面的剔除
      socket.broadcast.emit('systemMsg', socket.nickname + '离开！'); //广播除本人以外的其他人
    }
  });
});

//注意：这里要使用http，使用app监听端口的话，前台页面会不能访问到必须的js文件
http.listen(3000, function () {
  console.log('server start');
});

    在对应得聊天页面主要是引用相应的socket.io.js文件，创建一个socket，然后再监听发送相应的事件即可。

    <!DOCTYPE html>
<html>
  <head>
    <title></title>
    <link href="//cdn.bootcss.com/bootstrap/3.3.7/css/bootstrap.min.css" rel="stylesheet">

    <script src="//cdn.bootcss.com/jquery/3.1.1/jquery.min.js"></script>
    <script src="//cdn.bootcss.com/bootstrap/3.3.7/js/bootstrap.min.js"></script>
  </head>
  <body>
    <div class="container">
      <div id="messages">
      </div>
      <div class="sendBoard">
        <textarea class="message" placeHolder="输入消息"></textarea>
        <input class="btn btn-send" type="button" value="发送">
      </div>
    </div>
    <div class="loginBoard">
      <label class="info">连接服务器</label>
      <div class="user-info">
        <input type="text" placeHolder="nickname" name="nickname" />
        <input type="button" value="OK" id="loginBtn" />
      </div>
    </div>
    <script src="/socket.io/socket.io.js"></script>
    <script>
      var socket = io();
      socket.on('connect', function() { //当客户端连接上服务器的时候
        $('.loginBoard .info').text('输入你的昵称！');
        $('.loginBoard .user-info').show();
        $('.loginBoard input[type=text]').focus();
      });
      $('.loginBoard input[type=button]').click(function(event) { //发送登录信息到服务器
        var nickname = $('.loginBoard input[type=text]').val();
        socket.emit('login', nickname);
      });
      $('.sendBoard .btn-send').click(function(event) { //发送消息到服务器
        var msg = $('.sendBoard div.message')[0].innerHTML;
        if (msg.length != 0) {
          socket.emit('postMsg', msg);
          $('.sendBoard div.message')[0].innerHTML = '';
        };
      });

        self.socket.on('loginFail', function(msg) {  //监听登录失败事件
          $('.loginBoard .info').text(msg);
        });
        self.socket.on('loginSuccess', function (msg) { //登录成功
          $('.loginBoard').hide();
        });
        self.socket.on('systemMsg', function (msg) { //系统消息
          $('#historyMsgs').append('<p>系统消息：' + msg + '</p>');
        });
        self.socket.on('newMsg', function (user, msg) {
          $('#historyMsgs').append('<p>' + user + '：' + msg + '</p>');
        });
    </script>
  </body>
</html>

    # 结合express路由使用

    以使用express-generator创建的应用为例，把聊天功能放在某个路由下面

    对于app.js文件

    var io = require('socket.io')();

app.io = io;

//这部分最开始是放在路由里面的，但是发现很严重的问题：每次刷新页面不会触发disconnect事件，导致刷新几次后用户发消息会一次发送多条。
io.on('connection', function(socket){
  //在这里面监听和发送相应的事件
  socket.on('init', function () { //这个是为下面使用session验证使用的
    if (socket.handshake.headers.sessions) {
      socket.emit('loginSuccess');
      io.sockets.emit('systemMsg', socket.nickname + '加入房间！');
    } else {
      return socket.emit('loginFail', '请先登录！');
    }
  });
  socket.on('error', (error) => {
    console.log(error)
  });
  socket.on('disconnect', function(reason) {
    if (socket.nickname) {
      socket.broadcast.emit('systemMsg', socket.nickname + '离开！');
    }
  });
});

    # 结合session登录验证

    在这里要做的就是让express和socket.io共享session，一个比较简单的方案就是把用户的session存储到数据库当中，然后根据用户的cookie去查找相应的session。还可以选择直接把session加密存储在cookie当中，缺点就是每次请求都会额外增加很多东西。最后还是决定使用redis来缓存。
    在express中使用redis比较方便，直接安装引用connect-redis就行了。

    var session = require('express-session');
var RedisStore = require('connect-redis')(session);

var sessionStore = new RedisStore({
  "host": "127.0.0.1",
  "port": "6379",
});

app.use(session({
  name: config.session.key,
  secret: config.session.secret,
  resave: true,
  saveUninitialized: true,
  store: sessionStore,  //设置session的存储方式
  cookie: {
    maxAge: config.session.maxAge
  }
}));


//如果回调函数返回的第二个参数是false，那么客户端就连接不上服务器
io.set('authorization', function(handshakeData, cb) {
  var cookies = cookie.parse(handshakeData.headers.cookie);
  var connectKey = cookies[config.session.key];
  if (connectKey) {
    var connected = cookieParser.signedCookie(connectKey, config.session.secret);
    if(connected) { //前面都是为了得到这个session的id
      sessionStore.client.get('sess:' + connected, function (error, session) { //从redis获取对应session
        if (error) {
          cb(error, false);
        } else {
          handshakeData.headers.sessions = JSON.parse(session);
          if (JSON.parse(session).user) {
            cb(null, true);
          } else {  
            cb('未登录', false);
          }
        }
      });
    } else {
      cb('没有该Session', false);
    }
  }
});
