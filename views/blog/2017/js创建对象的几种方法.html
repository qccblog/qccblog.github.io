    js创建对象的几种方法

    qiucong 2017/4/29 js基础梳理

    总结了一些常见的创建对象的方法。

    # 原始模式

    var person = new Object();
person.name = "zhang";
person.show = function () {
    console.log(this.name);
}
person.show();

    需要使用的时候就单独创建一个对象，再添加属性和方法。不同的实例需要单独创建。

    # 工厂模式

    function Parent(name, age) {
    var person = new Object();
    person.name = name;
    person.age = age;
    person.show = function () {
        console.log("我是" + this.name);
    }
    return person;
}

var child = new Parent('zhang', 18);
child.show();

    将各种方法封装起来，返回整个对象。其中函数对象可以移到外面，让多个对象共享函数对象。

    # 构造函数

    function Parent(name, age) {
    this.name = name;
    this.age = age;
    this.show = function () {
        console.log("我是" + this.name);
    }
}

var child = new Parent('zhang', 18);
child.show();

    在构造函数内使用 this 关键字,创建属性和方法。再使用 new 运算符创建实例，通过传参生成不同的实例。构造函数会重复生成函数，为每个对象都创建独立的函数版本

    # 原型方式

    function Parent() {
}
Parent.prototype.name = "zhang";
Parent.prototype.like = new Array();
Parent.prototype.show = function () {
    console.log(this.like);
}

var child1 = new Parent();
var child2 = new Parent();
child1.like.push("zhangsan");
child1.like.push("lisi");
child1.show();

    # 混合的构造函数/原型方式

    function Parent(name, age) {
    this.name = name;
    this.age = age;
}

Parent.prototype.show = function () {
    console.log("我是" + this.name);
}
var child = new Parent('zhang', 18);

    利用构造函数方式创建函数的不同的部分，利用原型方式来定义相同的方法和属性。此种 方式即符合语义又可以扩展新的属性和方法

    # 动态原型方式

    function Parent()
{
    this.username = "zhang";
    if(typeof Parent.flag == "undefined") {
        console.log("create");
        Parent.prototype.show = function()
        {
            console.log(this.username);
        }
        Parent.flag = true;
    }
}

var child = new Parent();
var child2 = new Parent();

child.show();
child2.show();
