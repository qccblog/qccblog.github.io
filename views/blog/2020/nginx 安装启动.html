    # Linux 安装

    # yum 安装

    # 安装 nginx
sudo yum -y install nginx

# 卸载 nginx
sudo yum remove nginx

    新建一个下载源文件 /etc/yum.repos.d/nginx.repo。

    [nginx-stable]
name=nginx stable repo
baseurl=http://nginx.org/packages/centos/$releasever/$basearch/
gpgcheck=1
enabled=1
gpgkey=https://nginx.org/keys/nginx_signing.key
module_hotfixes=true

[nginx-mainline]
name=nginx mainline repo
baseurl=http://nginx.org/packages/mainline/centos/$releasever/$basearch/
gpgcheck=1
enabled=0
gpgkey=https://nginx.org/keys/nginx_signing.key
module_hotfixes=true

    使用 yum 进行 Nginx 安装时，Nginx 配置文件在/etc/nginx 目录下。

    # Ubuntu 安装

    # 安装依赖
sudo apt install curl gnupg2 ca-certificates lsb-release

# 设置稳定版的下载源
echo "deb http://nginx.org/packages/ubuntu `lsb_release -cs` nginx" \
    | sudo tee /etc/apt/sources.list.d/nginx.list
    
# 导入官方key验证下载的安装包
curl -fsSL https://nginx.org/keys/nginx_signing.key | sudo apt-key add -

# 可以验证一下
sudo apt-key fingerprint ABF5BD827BD9BF62

# 结果应该如下
pub   rsa2048 2011-08-19 [SC] [expires: 2024-06-14]
      573B FD6B 3D8F BC64 1079  A6AB ABF5 BD82 7BD9 BF62
uid   [ unknown] nginx signing key <signing-key@nginx.com>

# 安装
sudo apt update
sudo apt install nginx

    # 源码包安装

    Nginx 源码包安装方式步骤比较繁琐，并且需要提前安装一些 Nginx 依赖库。

    # 安装 gcc 环境
sudo yum -y install gcc gcc-c++ # nginx编译时依赖gcc环境

# 安装 pcre
sudo yum -y install pcre pcre-devel # 让nginx支持重写功能

# 安装 zlib
# zlib库提供了很多压缩和解压缩的方式，nginx使用zlib对http包内容进行gzip压缩
sudo yum -y install zlib zlib-devel 

# 安装 openssl
# 安全套接字层密码库，用于通信加密
sudo yum -y install openssl openssl-devel

# 下载
wget http://nginx.org/download/nginx-1.16.1.tar.gz

# 解压缩
sudo tar -zxvf nginx-1.16.1.tar.gz

cd nginx-1.16.1

# 检查平台安装环境
/usr/local/nginx是nginx编译安装的目录（推荐），安装完后会在此目录下生成相关文件
./configure --prefix=/usr/local/nginx

# 如果前面的依赖库都安装成功后，执行./configure --prefix=/usr/local/nginx命令会显示一些环境信息。如果出现错误，一般是依赖库没有安装完成，可按照错误提示信息进行所缺的依赖库安装。

# 编译
make

# 安装
make install

    # 启动

    源码包安装与直接安装的 nginx 服务操作命令也不同。

    # 启动
/usr/local/nginx/sbin/nginx

# 重新加载服务
/usr/local/nginx/sbin/nginx -s reload

# 停止服务
/usr/local/nginx/sbin/nginx -s stop
